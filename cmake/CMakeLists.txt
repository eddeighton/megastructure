

cmake_minimum_required( VERSION 3.1...3.16 )

project( megastructure VERSION 1.0 LANGUAGES CXX )

#insist on ALL projects using c++ 17
set( CMAKE_CXX_STANDARD 17 )

######################################
#win32 specific settings
IF( ${WIN32} )

add_definitions(-D_WIN32_WINNT=0x0601)
add_definitions(-DNOMINMAX -D_NOMINMAX)

ENDIF( ${WIN32} )

#setup some common directories from the root of the project
set( CMAKE_PARENT_PATH ${CMAKE_CURRENT_LIST_DIR}/.. )
find_path( MEGA_ROOT_DIR NAMES api PATHS ${CMAKE_PARENT_PATH} )
set( MEGA_API_DIR ${MEGA_ROOT_DIR}/api )
set( MEGA_SRC_DIR ${MEGA_ROOT_DIR}/src )

#helper path if third party libs are in parent folder
set( MEGA_THIRD_PARTY_DIR ${MEGA_ROOT_DIR}/../../thirdparty_x64 )

include_directories( ${MEGA_API_DIR} )
include_directories( ${MEGA_SRC_DIR} )

#attempt to ensure only Debug and Release configs are generated by default
set(CMAKE_CONFIGURATION_TYPES RelWithDebInfo CACHE STRING INTERNAL FORCE)

#enable folders within generated IDE build systems
set_property(GLOBAL PROPERTY USE_FOLDERS ON)

#ensure language runtimes and other system libs are copied to installation folder
include (InstallRequiredSystemLibraries)

install( FILES ${MEGA_API_DIR}/egcomponent/egcomponent.hpp DESTINATION include/egcomponent )
install( FILES ${MEGA_API_DIR}/egcomponent/traits.hpp DESTINATION include/egcomponent )
#now simply add all the actual targets

add_subdirectory(component)
add_subdirectory(core)
add_subdirectory(driver)
add_subdirectory(hosts)
add_subdirectory(master)
add_subdirectory(protocol)
add_subdirectory(python_lib)
add_subdirectory(schema)
add_subdirectory(slave)
add_subdirectory(tests)
add_subdirectory(unreal)
add_subdirectory(resource_lib)
add_subdirectory(resid)
